SYMBOLS = ["EURUSD", "BTCUSD"]
TIMEFRAMES = ["1m", "5m", "30m", "1h", "4h", "1d", "1w"]

async def background_loop():

        for symbol in SYMBOLS:

def get_signal(symbol: str):

import numpy as np
from sklearn.linear_model import LinearRegression

def calculate_slope(series, window=10):
    y = series[-window:].values.reshape(-1, 1)
    x = np.arange(window).reshape(-1, 1)

    model = LinearRegression().fit(x, y)
    return model.coef_[0][0]

def add_slopes(df):
    for col in df.columns:
        if "ema_" in col:
            df[f"{col}_slope"] = df[col].rolling(15).apply(
                lambda x: calculate_slope(x), raw=False
            )
    return df

🧠 5. DECISION ENGINE (EL CEREBRO)
engine/decision.py
def ema_alignment_score(row):
    emas = [row[f'ema_{p}'] for p in [3,5,10,25,80,180,220]]
    return all(x > y for x, y in zip(emas, emas[1:]))

def slope_strength(row):
    slopes = [row[f'ema_{p}_slope'] for p in [3,5,10]]
    return sum(slopes)

def alligator_state(row):
    if row['lips'] > row['teeth'] > row['jaw']:
        return "bullish"
    elif row['lips'] < row['teeth'] < row['jaw']:
        return "bearish"
    return "range"

def decision(row):
    score = 0

    if ema_alignment_score(row):
        score += 3

    if slope_strength(row) > 0:
        score += 2

    if alligator_state(row) == "bullish":
        score += 1

    if score >= 5:
        return "BUY", score
    elif score <= -3:
        return "SELL", score

    return "HOLD", score
📡 6. GENERADOR DE SEÑALES
engine/signals.py
from engine.multi_tf import analyze_timeframe
from engine.decision import decision

def generate_signal(symbol, timeframes):
    results = {}

    confirmations = []

    for tf in timeframes:
        row = analyze_timeframe(symbol, tf)
        action, score = decision(row)

        results[tf] = {
            "action": action,
            "score": score
        }

        if tf in ["4h", "1d", "1w"]:
            confirmations.append(action)

    # validación multi-TF
    if confirmations.count("BUY") >= 2:
        final = "BUY"
    elif confirmations.count("SELL") >= 2:
        final = "SELL"
    else:
        final = "HOLD"

    return {
        "symbol": symbol,
        "final_signal": final,
        "details": results
    }

🧠 IDEA CLAVE
Cada EMA puede aproximarse localmente como una recta:

EMA(t) ≈ m·t + b
Donde:

m = pendiente (velocidad del precio suavizado)

b = intercepto

👉 El cruce entre dos EMAs ocurre cuando:

EMA_fast(t) = EMA_slow(t)
📐 SOLUCIÓN MATEMÁTICA
Si tienes:

EMA_f(t) = m1·t + b1
EMA_s(t) = m2·t + b2
El punto de cruce es:

t_cross = (b2 - b1) / (m1 - m2)
Y el precio en ese cruce:

price_cross = m1 * t_cross + b1

import numpy as np
from sklearn.linear_model import LinearRegression

def fit_line(series, window=20):
    y = series[-window:].values.reshape(-1, 1)
    x = np.arange(window).reshape(-1, 1)

    model = LinearRegression().fit(x, y)

    slope = model.coef_[0][0]
    intercept = model.intercept_[0]

    return slope, intercept


def predict_cross(series_fast, series_slow, window=20):
    m1, b1 = fit_line(series_fast, window)
    m2, b2 = fit_line(series_slow, window)

    # evitar división por cero
    if abs(m1 - m2) < 1e-6:
        return None

    t_cross = (b2 - b1) / (m1 - m2)

    # ignorar cruces en pasado
    if t_cross < 0:
        return None

    price_cross = m1 * t_cross + b1

    return {
        "candles_to_cross": float(t_cross),
        "price": float(price_cross),
        "slopes": (m1, m2)
    }

import numpy as np
from engine.projection import predict_cross

EMA_PAIRS = [(3,5),(5,10),(10,25),(25,80),(80,180)]

TF_WEIGHTS = {
    "1m": 1,
    "5m": 2,
    "30m": 3,
    "1h": 4,
    "4h": 6,
    "1d": 8,
    "1w": 10
}

def project_timeframe(df, tf):
    projections = []

    for fast, slow in EMA_PAIRS:
        res = predict_cross(df[f'ema_{fast}'], df[f'ema_{slow}'])

        if res:
            res["tf"] = tf
            res["pair"] = f"{fast}-{slow}"
            res["weight"] = TF_WEIGHTS[tf]
            projections.append(res)

    return projections

TF_TO_MIN = {
    "1m": 1,
    "5m": 5,
    "30m": 30,
    "1h": 60,
    "4h": 240,
    "1d": 1440,
    "1w": 10080
}

def normalize_projection(p):
    minutes = p["candles_to_cross"] * TF_TO_MIN[p["tf"]]
    p["time_minutes"] = minutes
    return p
🧠 3. CONSENSO DE CRUCES
Agrupar predicciones cercanas en tiempo → “evento”

def cluster_projections(projections, tolerance=60):
    clusters = []

    for p in projections:
        assigned = False

        for c in clusters:
            if abs(c["center"] - p["time_minutes"]) < tolerance:
                c["points"].append(p)
                c["center"] = np.mean([x["time_minutes"] for x in c["points"]])
                assigned = True
                break

        if not assigned:
            clusters.append({
                "center": p["time_minutes"],
                "points": [p]
            })

    return clusters
🧠 4. SCORING DE EVENTO
def score_cluster(cluster):
    score = 0

    for p in cluster["points"]:
        score += p["weight"]

        # cruces cercanos pesan más
        if p["candles_to_cross"] < 10:
            score += 2

    return score
🧠 5. FILTRO ANTI-RUIDO
Descartar basura:

def filter_clusters(clusters, min_score=10, min_points=2):
    valid = []

    for c in clusters:
        if len(c["points"]) >= min_points:
            score = score_cluster(c)
            if score >= min_score:
                c["score"] = score
                valid.append(c)

    return valid
🧠 6. DIRECCIÓN DEL EVENTO
def infer_direction(cluster):
    bullish = 0
    bearish = 0

    for p in cluster["points"]:
        m1, m2 = p["slopes"]

        if m1 > m2:
            bullish += 1
        else:
            bearish += 1

    if bullish > bearish:
        return "BUY"
    elif bearish > bullish:
        return "SELL"

    return "NEUTRAL"

def analyze_projections(all_tf_data):
    all_projections = []

    for tf, df in all_tf_data.items():
        p = project_timeframe(df, tf)
        p = [normalize_projection(x) for x in p]
        all_projections.extend(p)

    clusters = cluster_projections(all_projections)
    clusters = filter_clusters(clusters)

    results = []

    for c in clusters:
        results.append({
            "time_minutes": c["center"],
            "score": c["score"],
            "direction": infer_direction(c),
            "details": c["points"]
        })

    return sorted(results, key=lambda x: -x["score"])

def calculate_sl_tp(row, direction, rr=2.0):
    atr = row['atr']
    price = row['close']

    if direction == "BUY":
        sl = price - atr * 1.5
        tp = price + (price - sl) * rr

    elif direction == "SELL":
        sl = price + atr * 1.5
        tp = price - (sl - price) * rr

    else:
        return None, None

    return sl, tp
💰 POSITION SIZING (CLAVE REAL)
def position_size(balance, risk_pct, entry, sl):
    risk_amount = balance * risk_pct

    distance = abs(entry - sl)
    if distance == 0:
        return 0

    size = risk_amount / distance
    return size

def run_backtest(df):

    entries = []
    exits = []

    for i in range(len(df)):
        row = df.iloc[i]

        signal = generate_signal("TEST", ["1h"])

        if signal["signal"] == "BUY":
            entries.append(True)
            exits.append(False)

        elif signal["signal"] == "SELL":
            entries.append(False)
            exits.append(True)

        else:
            entries.append(False)
            exits.append(False)

    pf = vbt.Portfolio.from_signals(
        df['close'],
        entries,
        exits,
        init_cash=10000
    )

    return pf.stats()

import gym
import numpy as np

class TradingEnv(gym.Env):

    def __init__(self, data):
        self.data = data
        self.current = 0

    def reset(self):
        self.current = 0
        return self._get_obs()

    def _get_obs(self):
        row = self.data.iloc[self.current]
        return np.array([
            row['ema_3'],
            row['ema_10'],
            row['ema_3_slope']
        ])

    def step(self, action):
        reward = 0

        if action == 1:  # BUY
            reward = self.data.iloc[self.current+1]['close'] - self.data.iloc[self.current]['close']

        elif action == 2:  # SELL
            reward = self.data.iloc[self.current]['close'] - self.data.iloc[self.current+1]['close']

        self.current += 1
        done = self.current >= len(self.data) - 1

        return self._get_obs(), reward, done, {}
